Создание БД:
CREATE DATABASE `lybrary`

Создание таблиц:
CREATE TABLE `author` (
  `author_id` int NOT NULL AUTO_INCREMENT,
  `last_name` text,
  `first_name` text,
  `middle_name` text,
  `original_lang_id` int DEFAULT NULL,
  `full_name_original` text,
  `short_name_rus` text,
  `short_name_original` text,
  PRIMARY KEY (`author_id`)
)

CREATE TABLE `book` (
  `Book_name_rus` text NOT NULL,
  `book_id` int NOT NULL AUTO_INCREMENT,
  `original_book_name` text,
  `author_id` int DEFAULT NULL,
  `publisher` text,
  `translate` text,
  `genre_id` int DEFAULT NULL,
  `year` smallint DEFAULT NULL,
  `binding_type` text,
  `shelf_number` int DEFAULT NULL,
  `page_count` smallint DEFAULT NULL,
  PRIMARY KEY (`book_id`)
  )

  CREATE TABLE `language` (
  `language_id` int NOT NULL AUTO_INCREMENT,
  `language name` text NOT NULL,
  PRIMARY KEY (`language_id`)
  )

  CREATE TABLE `genre` (
  `genre_id` int NOT NULL AUTO_INCREMENT,
  `genre name` text NOT NULL,
  PRIMARY KEY (`genre_id`)
)

CREATE TABLE `shelf` (
  `shilf number` int NOT NULL AUTO_INCREMENT,
  `shelving` smallint NOT NULL,
  `shelf` smallint NOT NULL,
  PRIMARY KEY (`shilf number`),
  UNIQUE KEY `shilf number` (`shilf number`)
)
